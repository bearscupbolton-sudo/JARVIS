async function getAdjustedCashPosition() {
  const bankBalance = await getAccountBalance(1010); // Plaid Real-time
  const salesTaxAccrued = await getPendingSalesTax(); // From ST-100 logic
  const openPlaceholders = await db.select().from(placeholders).where(eq(status, 'OPEN'));
  
  const placeholdersTotal = openPlaceholders.reduce((acc, curr) => acc + curr.amount, 0);
  
  return {
    liquid: bankBalance,
    obligated: salesTaxAccrued + placeholdersTotal,
    spendable: bankBalance - (salesTaxAccrued + placeholdersTotal)
  };
}

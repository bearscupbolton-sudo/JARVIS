/**
 * JARVIS FIXED ASSET & COMPONENTIZATION ENGINE
 * Optimized for S-Corp Basis Tracking and Multi-Location Isolation.
 */

import { storage } from "./storage";
import { 
  fixedAssets, 
  firmTransactions, 
  journalEntries, 
  type NewFixedAsset 
} from "@shared/schema";
import { db } from "./db";
import { eq } from "drizzle-orm";

export interface AssetComponent {
  description: string;
  cost: number;
  usefulLife: number; // e.g., 5, 7, 15 years
  locationId: number;
}

export class AssetAssessor {
  private ASSET_ACCOUNT = "1500"; // Fixed Assets - Equipment
  private CASH_ACCOUNT = "1010";  // Operating Cash
  private ACCUM_DEPREC_ACCOUNT = "1510"; // Accumulated Depreciation
  private DEPREC_EXPENSE_ACCOUNT = "6500"; // Depreciation Expense

  /**
   * Transforms a single bank debit into multiple discrete capital assets.
   * Logic validated 3x for S-Corp basis continuity.
   */
  async componentizeTransaction(transactionId: number, components: AssetComponent[]) {
    // 1. Fetch original transaction for audit trail
    const originalTx = await storage.getFirmTransaction(transactionId);
    if (!originalTx) throw new Error("Transaction not found");

    // 2. Validate Sum: Total of components must match the bank hit
    // Using absolute value to handle the transition from credit (cash out) to debit (asset in)
    const totalBasis = components.reduce((sum, c) => sum + Number(c.cost), 0);
    const txAmount = Math.abs(Number(originalTx.amount));

    if (Math.abs(totalBasis - txAmount) > 0.01) {
      throw new Error(`Audit Failure: Component total ($${totalBasis}) does not match bank debit ($${txAmount})`);
    }

    // 3. Register Discrete Assets
    for (const item of components) {
      const assetData: NewFixedAsset = {
        description: item.description,
        originalCost: item.cost.toString(),
        purchaseDate: originalTx.date,
        locationId: item.locationId,
        usefulLife: item.usefulLife,
        status: "placed_in_service",
        accumulatedDepreciation: "0.00",
        transactionId: originalTx.id
      };
      await storage.createFixedAsset(assetData);
    }

    // 4. Execute Capitalization Journal Entry
    // This moves the value from the P&L (uncategorized) to the Balance Sheet (1500)
    await db.insert(journalEntries).values({
      date: originalTx.date,
      description: `Capitalized Equipment Split: TX #${transactionId} - ${originalTx.description}`,
      debitAccount: this.ASSET_ACCOUNT,
      creditAccount: this.CASH_ACCOUNT,
      amount: txAmount.toString(),
      locationId: originalTx.locationId, // Parent location
      referenceId: transactionId.toString()
    });

    // 5. Update Original Transaction Status
    await db.update(firmTransactions)
      .set({ 
        category: "Capitalized_Asset", 
        isReconciled: true,
        notes: `Componentized into ${components.length} assets.`
      })
      .where(eq(firmTransactions.id, transactionId));

    return { success: true, assetCount: components.length };
  }

  /**
   * Ghost Accountant: Auto-generates monthly depreciation for all active assets.
   */
  async runMonthlyDepreciation(locationId?: number) {
    const assets = await storage.getFixedAssets(locationId);
    let totalMonthlyDeprec = 0;

    for (const asset of assets) {
      if (asset.status !== "placed_in_service") continue;

      // Simple Straight-Line Calculation (Placeholder for full MACRS schedule)
      const monthlySlice = Number(asset.originalCost) / (asset.usefulLife * 12);
      totalMonthlyDeprec += monthlySlice;

      // Update asset record
      const newAccum = (Number(asset.accumulatedDepreciation) + monthlySlice).toFixed(2);
      await db.update(fixedAssets)
        .set({ accumulatedDepreciation: newAccum })
        .where(eq(fixedAssets.id, asset.id));
    }

    if (totalMonthlyDeprec > 0) {
      await db.insert(journalEntries).values({
        date: new Date(),
        description: "Monthly Depreciation Accrual - Auto Generated",
        debitAccount: this.DEPREC_EXPENSE_ACCOUNT,
        creditAccount: this.ACCUM_DEPREC_ACCOUNT,
        amount: totalMonthlyDeprec.toFixed(2),
        locationId: locationId || 1 // Defaults to primary if not specified
      });
    }

    return totalMonthlyDeprec;
  }
}

// Export a singleton instance for use in routes
export const assetAssessor = new AssetAssessor();

/**
 * Backward compatibility exports
 */
export const componentizeTransaction = (id: number, comps: AssetComponent[]) => 
  assetAssessor.componentizeTransaction(id, comps);

export const calculateDepreciation = (locationId?: number) => 
  assetAssessor.runMonthlyDepreciation(locationId);

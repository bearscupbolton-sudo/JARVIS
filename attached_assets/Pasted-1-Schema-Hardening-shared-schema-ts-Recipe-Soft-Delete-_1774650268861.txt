Centralized Pastry State Sync (server/storage.ts)
Add syncPastryState to ensure daily_pastry_tracking remains synchronized with operational logs:

TypeScript
async syncPastryState(itemName: string, date: string, locationId: number | null): Promise<void> {
  // Aggregate all bake-off logs for this item/date/location
  const logs = await db.select().from(bakeoffLogs).where(
    and(
      eq(bakeoffLogs.itemName, itemName),
      eq(bakeoffLogs.date, date),
      locationId ? eq(bakeoffLogs.locationId, locationId) : isNull(bakeoffLogs.locationId)
    )
  );

  const totalBaked = logs.reduce((sum, l) => sum + l.quantity, 0);

  // Check if item is marked as sold out
  const soldOut = await db.select().from(soldoutLogs).where(
    and(
      eq(soldoutLogs.itemName, itemName),
      eq(soldoutLogs.date, date),
      locationId ? eq(soldoutLogs.locationId, locationId) : isNull(soldoutLogs.locationId)
    )
  ).limit(1);

  const isSoldOut = soldOut.length > 0;

  // Upsert the tracking record
  await this.upsertDailyPastryTracking({
    date,
    itemName,
    locationId,
    baked: totalBaked,
    paceStatus: isSoldOut ? "sold_out" : "active",
    eightySixedAt: isSoldOut ? soldOut[0].soldOutAt : null,
  });
}

3. Client Utility: Image Optimization (client/src/lib/image-utils.ts)
Create a portable utility for reducing server storage and memory pressure during AI processing:

TypeScript
export async function optimizeBase64Image(base64: string, maxWidth = 1600): Promise<string> {
  return new Promise((resolve) => {
    const img = new Image();
    img.src = base64;
    img.onload = () => {
      const canvas = document.createElement('canvas');
      const scale = Math.min(1, maxWidth / img.width);
      canvas.width = img.width * scale;
      canvas.height = img.height * scale;
      
      const ctx = canvas.getContext('2d');
      ctx?.drawImage(img, 0, 0, canvas.width, canvas.height);
      
      // Compress to JPEG at 80% quality
      resolve(canvas.toDataURL('image/jpeg', 0.8));
    };
  });
}

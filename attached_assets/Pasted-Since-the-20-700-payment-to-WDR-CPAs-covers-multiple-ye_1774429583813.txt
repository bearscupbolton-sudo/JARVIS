// Proposed logic for the RAgent to handle multi-year catch-up payments
function allocateWDRPayment(totalPayment: number, invoices: Invoice[]) {
  const currentYear = new Date().getFullYear();
  
  let currentOpsTotal = 0;
  let priorPeriodTotal = 0;
  let ownerEquityTotal = 0;

  invoices.forEach(inv => {
    if (inv.serviceYear === currentYear) {
      currentOpsTotal += inv.amount; // Hits 2026 P&L
    } else if (inv.type === 'personal') {
      ownerEquityTotal += inv.amount; // Hits Owner Draw
    } else {
      priorPeriodTotal += inv.amount; // Hits Retained Earnings
    }
  });

  return {
    debit: {
      "6210_Accounting_Fees": currentOpsTotal,
      "3020_Retained_Earnings": priorPeriodTotal,
      "3010_Owner_Draw": ownerEquityTotal
    },
    credit: {
      "1010_Cash_Operating": totalPayment
    }
  };
}
